payon_in02,64,71,4	script	Archer Guildsman#archer	85,{
	if (Upper == 1) {
		if (Class == Job_Novice_High && (ADVJOB == Job_Sniper || ADVJOB == Job_Clown || ADVJOB == Job_Gypsy)) {
			mes "[Homme de la Guilde des Archers]";
			mes "Hé, je te connais.";
			mes "Tu as passé ce test";
			mes "auparavant, n'est-ce pas ?";
			next;
			mes "[Homme de la Guilde des Archers]";
			mes "Ah, tu dois avoir été";
			mes "à Valhalla et être rené.";
			mes "Wow, c'est tellement impressionnant !";
			next;
			if (!callfunc("F_CanChangeJob")) {
				mes "[Homme de la Guilde des Archers]";
				mes "Euh...";
				mes "Tu ferais bien d'apprendre d'abord toutes les Compétences de Base avant de devenir un Archer.";
				next;
				mes "[Homme de la Guilde des Archers]";
				mes "D'accord, à plus tard.";
				close;
			}
			mes "[Homme de la Guilde des Archers]";
			mes "Bon, je n'ai";
			mes "rien d'autre à dire.";
			mes "Je sais que tu feras un excellent Archer...";
			next;
			skill "NV_TRICKDEAD", 0, SKILL_PERM;
			jobchange Job_Archer_high;
			atcommand "@joblvlup 5";
			skill "AC_MAKINGARROW", 1, SKILL_PERM;
			skill "AC_CHARGEARROW", 1, SKILL_PERM;
			mes "[Homme de la Guilde des Archers]";
			mes "Bien qu'il n'y ait pas de récompense spéciale pour toi cette fois-ci, j'espère que tu comprends. Prends soin de toi.";
			close;
		} else {
			mes "[Homme de la Guilde des Archers]";
			mes "Oh... ?";
			mes "Hé, que fais-tu";
			mes "ici... ?";
			next;
			mes "[Homme de la Guilde des Archers]";
			mes "Je peux dire que tu n'es pas fait pour être un Archer. On dirait que tu es destiné à faire";
			mes "quelque chose d'autre...";
			close;
		}
	}
	mes "[Homme de la Guilde des Archers]";
	mes "Enchanté. Comment puis-je t'aider ?";
	next;
	switch (select("Je veux être un Archer.:Je veux connaître les prérequis, s'il te plaît.:Rien, merci.")) {
		case 1:
			mes "[Homme de la Guilde des Archers]";
			if (BaseJob != Job_Novice) {
				if (BaseJob == Job_Archer)
					mes "Haha, tu plaisantes...";
				else {
					mes "Je suis désolé, mais seuls les Novices peuvent changer de job.";
					mes "Tu as déjà un bon job, n'est-ce pas ?";
				}
				close;
			} else {
				mes "Tu es... " + strcharinfo(0) + ", c'est ça ?";
				next;
				if (!callfunc("F_CanChangeJob")) {
					mes "[Homme de la Guilde des Archers]";
					mes "Eh bien, tu n'as pas le bon niveau de compétence.";
					mes "Ton niveau de job doit être d'au moins ^4A4AFF10^000000 et ton niveau de Compétence de Base doit atteindre ^4A4AFFle niveau 9";
					next;
					mes "[Homme de la Guilde des Archers]";
					mes "Parce qu'un Archer a besoin d'une concentration extrêmement élevée, nous n'acceptons pas ceux qui ont peu de patience.";
					close;
				}
				mes "[Homme de la Guilde des Archers]";
				mes "Ton niveau de Compétence de Base est maintenant suffisant..";
				mes "....Hm~~ tu es maintenant prêt à devenir un Archer. Je vais passer à l'étape tout de suite.";
				next;
				mes "[Homme de la Guilde des Archers]";
				mes "Félicitations ! Tu es maintenant un Archer ! De plus, nous espérons que tu participeras activement à de nombreux programmes pour la relance de la Guilde des Archers.";
				mes "Ah, des objets sont arrivés du Département de Production. Tiens, prends ça ! Ce sont tous à toi !";
				callfunc "Job_Change", Job_Archer;
				atcommand "@joblvlup 5";
				getitem 1742, 1; // N_Composite_Bow
				getitem 12004, 1; // Arrow_Container
				getitem 12009, 1; // Silver_Arrow_Container
				getitem 12008, 1; // Fire_Arrow_Container
				next;
				mes "[Homme de la Guilde des Archers]";
				mes "Avec un arc et des flèches, tu es maintenant devenu un vrai Archer.";
				mes "Si tu ouvres le conteneur de flèches, il y a des flèches dedans et tu peux les équiper.";
				mes "Eh bien, j'attends de meilleures nouvelles de ta part. Il est temps de dire au revoir.";
				mes "Salut.";
				close;
			}
		case 2:
			mes "[Homme de la Guilde des Archers]";
			mes "Un Archer a des compétences avec un arc et a divers talents.";
			mes "La plus grande capacité d'un Archer est d'attaquer les ennemis de loin.";
			next;
			mes "[Homme de la Guilde des Archers]";
			mes "Bien qu'un Archer ait une HP plus faible, il peut tirer sur les ennemis à longue distance,";
			mes "donc un Archer est plus en sécurité dans un vrai combat.";
			next;
			mes "[Homme de la Guilde des Archers]";
			mes "Bien qu'un Archer dans Ragnarok ait une HP plus faible, il a une précision élevée et un taux d'attaque élevé de sorte que l'archer peut tuer les monstres avant qu'ils ne s'approchent d'un Archer.";
			next;
			mes "[Homme de la Guilde des Archers]";
			mes "^8C2121Un Archer peut changer de job pour devenir un Chasseur.^000000";
			mes "^8C2121En plus de Chasseur, si tu es un homme, tu peux changer de job pour devenir un Barde et si tu es une femme, tu peux changer de job pour devenir une Danseuse.^000000";
			close;
		case 3:
			mes "[Homme de la Guilde des Archers]";
			mes "Si tu as des questions, n'hésite pas à venir me les poser.";
			close;
	}
}
