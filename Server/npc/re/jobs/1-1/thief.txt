moc_prydb1,39,129,2	script	Thief Guide#thief	69,{
	if (Upper == 1) {
		if (ADVJOB == Job_Assassin_Cross || ADVJOB == Job_Stalker) {
			if (Class == Job_Novice_High) {
				mes "[Guide des Voleurs]";
				mes "Hein ? Est-ce que je te connais ? C'est étrange, tu me sembles tellement familier. Tu n'as pas un jumeau, n'est-ce pas ?";
				next;
				if (!callfunc("F_CanChangeJob")) {
					mes "[Guide des Voleurs]";
					mes "Quoi, tu veux devenir un Voleur ? Désolé, mais tu as l'air d'avoir besoin de plus d'entraînement.";
					next;
					mes "[Guide des Voleurs]";
					mes "Prends ton temps et apprends toutes les compétences de base, d'accord ? Eh bien, à plus tard~ !";
					close;
				}
				mes "[Guide des Voleurs]";
				mes "Eh bien, j'ai cette impression que tu as vécu une vie entière de combats, alors je te promeus au rang de Voleur dès maintenant. Je ferais mieux de donner aux durs ce qu'ils veulent...";
				next;
				skill "NV_TRICKDEAD", 0, SKILL_PERM;
				jobchange Job_Thief_High;
				atcommand "@joblvlup 5";
				skill "TF_SPRINKLESAND", 1, SKILL_PERM;
				skill "TF_BACKSLIDING", 1, SKILL_PERM;
				skill "TF_PICKSTONE", 1, SKILL_PERM;
				skill "TF_THROWSTONE", 1, SKILL_PERM;
				mes "[Guide des Voleurs]";
				mes "Maintenant que tu es devenu un Voleur, vis comme un Voleur. Allez, fonce ! Suivant~";
				close;
			} else {
				mes "[Guide des Voleurs]";
				if (Sex == SEX_MALE)
					mes "Hé, mec.";
				else
					mes "Hé, chérie~";
				close;
			}
		} else {
			mes "[Guide des Voleurs]";
			if (Sex == SEX_MALE)
				mes "Hé, mec.";
			else
				mes "Hé, chérie.";
			mes "...Hé ! Tu as l'air trop gentil pour vouloir être un Voleur !! Maintenant dégage, je suis occupé. Suivant !";
			close;
		}
	}
	if (q_job_thief == 1) {
		mes "[Membre de la Guilde des Voleurs]";
		mes "Eh bien... On dirait le "+(Sex == SEX_MALE?"fils":"fille")+" bien-aimé d'une famille riche, mais pourquoi es-tu ici dans cet endroit sale ?";
		next;
		mes "[Membre de la Guilde des Voleurs]";
		mes "Eh bien, ce n'est pas moi qui suis responsable de faire de toi un Voleur. J'accepte seulement les demandes, tu comprends ?";
		mes "Si tu veux devenir un Voleur, demande au gars aux yeux perçants à côté de moi.";
		close;
	}
	mes "[Membre de la Guilde des Voleurs]";
	if (BaseJob != Job_Novice) {
		if (BaseJob == Job_Thief) {
			mes "Eh bien~ si tu as des problèmes, viens m'en parler à tout moment, d'accord ?";
			close;
		} else {
			mes "Qu'est-ce que c'est que ça... ?";
			if (Sex == SEX_MALE) {
				mes "Hein... tu es un "+jobname(Class)+", imbécile !";
				next;
				mes "[Membre de la Guilde des Voleurs]";
				mes "Hé, frère.";
			} else {
				next;
				mes "[Membre de la Guilde des Voleurs]";
				mes "Hé, demoiselle.";
			}
			mes "Pourquoi es-tu ici ? Retourne à ta place~ retourne~~";
			close;
		}
	}
	mes "Eh bien... On dirait le "+(Sex == SEX_MALE?"fils":"fille")+" bien-aimé d'une famille riche, mais pourquoi es-tu ici dans cet endroit sale ?";
	next;
	if (select("Je veux être un Voleur. : Rien.") == 2) {
		mes "[Membre de la Guilde des Voleurs]";
		mes "Tu sais que tu ne peux pas être un Voleur sans une demande...";
		mes "Qu'est-ce qui te passe par la tête... ?";
		close;
	}
	mes "[Membre de la Guilde des Voleurs]";
	mes "Eh bien, es-tu si fier de cela ?";
	mes "Tu me dis si fièrement que tu veux être un Voleur ! Pourquoi n'irais-tu pas dans tous les villages pour te faire de la publicité en tant que voleur ?";
	mes "'Ha ha ha ! Va mettre une bannière qui dit 'Je serai un voleur fier qui vole les affaires des autres'.";
	next;
	mes "[Membre de la Guilde des Voleurs]";
	mes "Tu veux vraiment être un voleur ?";
	next;
	switch (select("Oui. : Non. : Et toi ?")) {
		case 1:
			mes "[Membre de la Guilde des Voleurs]";
			mes "Oh, vraiment... ? Hein... bon... Je vis et j'apprends à voir des gens étranges comme toi.";
			next;
			break;
		case 2:
			mes "[Membre de la Guilde des Voleurs]";
			mes "Alors pourquoi es-tu ici ? Tu penses pouvoir devenir un voleur si facilement ?";
			next;
			break;
		case 3:
			mes "[Membre de la Guilde des Voleurs]";
			mes "Eh.. ? moi ? moi ?";
			mes "Eh bien... je conviens bien d'être un voleur... caractéristiquement... Ça ne me dérange pas cette affaire stupide.";
			next;
			break;
	}
	mes "[Membre de la Guilde des Voleurs]";
	mes "Quoi qu'il en soit, dans le monde extérieur, ne dis jamais que tu veux être un Voleur !!";
	next;
	mes "[Membre de la Guilde des Voleurs]";
	mes "Alors, veux-tu postuler pour être un Voleur ?";
	next;
	if (select("Oui, je veux. : Non.") == 2) {
		mes "[Membre de la Guilde des Voleurs]";
		mes "Eh bien... fais ce que tu veux~ Va dans ta direction~";
		close;
	}
	mes "[Membre de la Guilde des Voleurs]";
	mes "D'accord, dis-moi ton nom. Hein ? Quoi ?";
	mes ""+strcharinfo(0)+". Alors "+strcharinfo(0)+", c'est ça... ?";
	next;
	if (!callfunc("F_CanChangeJob")) {
		mes "[Membre de la Guilde des Voleurs]";
		mes "Je peux voir ta forte volonté de devenir un Voleur...";
		mes "Mais seulement avec ta volonté, tu ne peux pas t'en sortir dans un vrai combat, n'est-ce pas ?";
		mes "Alors va et atteins au moins le Niveau 9 des compétences de base.";
		close;
	}
	mes "[Membre de la Guilde des Voleurs]";
	mes "Hmm.... J'ai trouvé des antécédents criminels intéressants de toi dans les données de l'agence de détectives. Eh bien, tu sembles avoir ce qu'il faut pour être un bon voleur.";
	mes ""+strcharinfo(0)+". Je déclare maintenant que tu as réussi l'entretien d'embauche !";
	next;
	mes "[Membre de la Guilde des Voleurs]";
	mes "Eh bien, ce n'est pas moi qui suis responsable de faire de toi un Voleur. J'accepte seulement les demandes, tu comprends ?";
	mes "Si tu veux devenir un Voleur, demande au gars à côté de moi.";
	set q_job_thief, 1;
	close;
}

moc_prydb1,42,133,2	script	Thief Guildsman#thief	118,{
	if (q_job_thief == 1) {
		mes "[Membre de la Guilde des Voleurs]";
		mes "D'accord. Tu as dû réussir l'entretien d'embauche, hein ?";
		mes "Bien. Je vais t'accepter.";
		mes "Commençons la cérémonie de changement de job de notre guilde !";
		next;
		mes "[Commandant de la Guilde des Voleurs]";
		mes "'"+strcharinfo(0)+".'";
		mes "'Pour ceux qui sont prêts, la lumière de la lune ouvrira la voie. Maintenant, tu es pleinement prêt, je t'autorise officiellement à devenir membre de la Guilde des Voleurs.'";
		callfunc "Job_Change", Job_Thief;
		atcommand "@joblvlup 5";
		set q_job_thief, 0;
		getitem 13041, 1;    // N_Main_Gauche
		next;
		mes "[Commandant de la Guilde des Voleurs]";
		mes "'Félicitations pour devenir un Voleur.'";
		mes "'Dorénavant, respecte les règles de notre guilde et sois un membre honorable.'";
		mes "'Si tu nous apportes de la disgrâce en enfreignant nos règles, tu ferais mieux de surveiller tes arrières.'";
		mes "'Quoi qu'il en soit, je m'attends à ce que tu sois un grand voleur.'";
		next;
		mes "[Brad]";
		mes "Heee~Yaaaa~ ! Félicitations ! Mon ami.";
		mes "Mon nom est 'Brad'. Je suis responsable des ressources humaines ici.";
		mes "Je ne suis pas sûr pour l'instant, mais tu auras plus de chances de me voir plus tard.";
		next;
		mes "[Brad]";
		mes "D'accord, j'ai fait ce que je pouvais pour toi, alors va ton chemin. Je suis assez occupé.";
		mes "On se reverra.";
		close;
	}
	if (BaseJob != Job_Novice) {
		if (BaseJob == Job_Thief) {
			mes "[Brad]";
			mes "Je n'ai pas d'événements spéciaux maintenant. Alors va ton chemin et reviens plus tard.";
		} else {
			mes "[Membre de la Guilde des Voleurs]";
			mes "Hé~ Hé~ Tu n'es ni novice ni voleur !";
			mes "Qu'est-ce que tu fais ici ? Tu n'es pas le bienvenu pour faire de cet endroit ta maison~ Hweeeee~ Dégage d'ici~";
		}
		close;
	}
	mes "[Membre de la Guilde des Voleurs]";
	mes "Ho ? Pourquoi un novice comme toi visite-t-il cet endroit ?";
	mes "Si tu es ici pour devenir un Voleur, demande à la dame au tempérament désagréable juste à côté de moi.";
	close;
}
